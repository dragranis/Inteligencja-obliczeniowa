titanic <- read.csv("TitanicSurvival.csv",header = TRUE, sep=",")
library(arules)
library(arulesViz)
rules <- apriori(titanic)
rules <- apriori(titanic,parameter=list(minlen=2,supp=0.005,conf=0.8))
inspect(rules)
rules.sorted <- sort(rules,by="support")
inspect(rules.sorted)
survived<-titanic
survived<- subset(survived,survived=='yes')
survived.rules <- apriori(survived,parameter=list(minlen=2,supp=0.6,conf=0.8))
inspect(survived.rules)
survived.rules <- apriori(survived,parameter=list(minlen=2,supp=0.4,conf=0.8))
inspect(survived.rules)
plot(survived.rules, method="graph", control=list(nodeCol="red", edgeCol="blue"))
survived.rules <- apriori(survived,parameter=list(minlen=2,supp=0.2,conf=0.8))
inspect(survived.rules)
plot(survived.rules, method="graph", control=list(nodeCol="red", edgeCol="blue"))
